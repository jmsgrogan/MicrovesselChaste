70 template<unsigned DIM>
71 std::vector<boost::shared_ptr<VesselNode<DIM> > > LatticeBasedSproutingRule<DIM>::GetSprouts(const std::vector<boost::shared_ptr<VesselNode<DIM> > >& rNodes)
72 {
73  if(!this->mpVesselNetwork)
74  {
75  EXCEPTION("A vessel network is required for this type of sprouting rule.");
76  }
77 
78  // Set up the output sprouts vector
79  std::vector<boost::shared_ptr<VesselNode<DIM> > > sprouts;
80 
81  // Loop over all nodes and randomly select sprouts
82  for(unsigned idx = 0; idx < rNodes.size(); idx++)
83  {
84  if(rNodes[idx]->GetNumberOfSegments() != 2)
85  {
86  continue;
87  }
88 
89  // Check we are not too close to the end of the vessel
90  if(this->mVesselEndCutoff > 0.0 * unit::metres)
91  {
92  if(rNodes[idx]->GetSegment(0)->GetVessel()->GetClosestEndNodeDistance(rNodes[idx]->rGetLocation())< this->mVesselEndCutoff)
93  {
94  continue;
95  }
96  if(rNodes[idx]->GetSegment(1)->GetVessel()->GetClosestEndNodeDistance(rNodes[idx]->rGetLocation())< this->mVesselEndCutoff)
97  {
98  continue;
99  }
100  }
101 
102  // Check we are not too close to an existing candidate
103  if(mTipExclusionRadius>0.0 * unit::metres)
104  {
105  bool too_close = false;
106  for(unsigned jdx=0; jdx<sprouts.size(); jdx++)
107  {
108  if(rNodes[idx]->GetDistance(sprouts[jdx]->rGetLocation()) < mTipExclusionRadius)
109  {
110  too_close = true;
111  }
112  }
113  if(too_close)
114  {
115  continue;
116  }
117  }
118 
119  double prob_tip_selection = this->mSproutingProbability*SimulationTime::Instance()->GetTimeStep()*BaseUnits::Instance()->GetReferenceTimeScale();
120  if (RandomNumberGenerator::Instance()->ranf() < prob_tip_selection)
121  {
122  sprouts.push_back(rNodes[idx]);
123  }
124  }
125  return sprouts;
126 }
