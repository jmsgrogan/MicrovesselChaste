123 template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
124 double MichaelisMentenSteadyStateDiffusionReactionPde<ELEMENT_DIM, SPACE_DIM>::ComputeNonlinearSourceTermPrime(const ChastePoint<SPACE_DIM>& rX, double u)
125 {
126  if(u<0.0)
127  {
128  return (this->mConstantInUTerm / mMichaelisMentenThreshold)*(1.0*unit::seconds);
129  }
130  else
131  {
132  units::quantity<unit::concentration> u_scaled = u*unit::mole_per_metre_cubed;
133  return (this->mConstantInUTerm * mMichaelisMentenThreshold)/((mMichaelisMentenThreshold + u_scaled)*(mMichaelisMentenThreshold + u_scaled))*(1.0*unit::seconds);
134  }

137 template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
138 units::quantity<unit::concentration_flow_rate> MichaelisMentenSteadyStateDiffusionReactionPde<ELEMENT_DIM, SPACE_DIM>::ComputeNonlinearSourceTerm(unsigned gridIndex,
139  units::quantity<unit::concentration> u)
