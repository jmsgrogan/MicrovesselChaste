63 template<unsigned DIM>
64 void MicrovesselSimulationModifier<DIM>::SetupSolve(AbstractCellPopulation<DIM,DIM>& rCellPopulation, std::string outputDirectory)
65 {
66  // Do set-up on the vascular tumour modifier
67  if(mpSolver)
68  {
69  mpSolver->SetupFromModifier(rCellPopulation, mCellPopulationReferenceLength, mCellPopulationReferenceConcentration, outputDirectory);
70 
71  // Do the first solver increment
72  mpSolver->Increment();
73  }
74  else
75  {
76  EXCEPTION("A MicrovesselSolver is required for this modifier.");
77  }
78 
79  rCellPopulation.Update();
80 
81  for (typename AbstractCellPopulation<DIM>::Iterator cell_iter = rCellPopulation.Begin();
82  cell_iter != rCellPopulation.End();
83  ++cell_iter)
84  {
85  // Store the cell's volume in CellData
86  cell_iter->GetCellData()->SetItem("oxygen", 0.0);
87  }
88 
89  // Update the cell data
90  UpdateCellData(rCellPopulation);
91 }
