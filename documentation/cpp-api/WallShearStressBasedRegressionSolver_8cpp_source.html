74 template<unsigned DIM>
75 void WallShearStressBasedRegressionSolver<DIM>::Increment()
76 {
77  if(!this->mpNetwork)
78  {
79  EXCEPTION("The regression solver needs an initial vessel network");
80  }
81 
82  std::vector<boost::shared_ptr<Vessel<DIM> > > vessels = this->mpNetwork->GetVessels();
83  for(unsigned idx=0;idx<vessels.size(); idx++)
84  {
85  // if wall shear stress of vessel is below threshold then start regression timer, unless it has already been started
86  if (vessels[idx]->GetFlowProperties()->GetWallShearStress() < mThresholdWss)
87  {
88  if (!(vessels[idx]->GetFlowProperties()->HasRegressionTimerStarted()) && !(vessels[idx]->GetFlowProperties()->HasVesselRegressed(this->mReferenceTime)))
89  {
90  // increment time that the vessel has had low wall shear stress
91  vessels[idx]->GetFlowProperties()->SetTimeUntilRegression(mMaxTimeWithLowWss, this->mReferenceTime);
92  }
93  }
94  else // otherwise rescue vessel
95  {
96  // wall shear stress above threshold so vessel is not regressing
97  vessels[idx]->GetFlowProperties()->ResetRegressionTimer();
98  }
99  }
100 
101  // iterate through all vessels and if regression flag is true then remove from the network
102  for(unsigned idx=0;idx<vessels.size(); idx++)
103  {
104  if (vessels[idx]->GetFlowProperties()->HasVesselRegressed(this->mReferenceTime))
105  {
106  this->mpNetwork->RemoveVessel(vessels[idx], true);
107  }
108  }
109 }
